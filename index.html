    async function fetchWithRetry(url, options, retries = 5) {

      const delays = [1000, 2000, 4000, 8000, 16000];

      for (let i = 0; i < retries; i++) {

        try {

          const response = await fetch(url, options);

          if (!response.ok) throw new Error(`HTTP ${response.status}`);

          return await response.json();

        } catch (error) {

          if (i === retries - 1) throw error;

          await new Promise(res => setTimeout(res, delays[i]));

        }

      }

    }

                {feedback.correct

                  ? <button onClick={nextTurn} style={{...btn("#3498db"),padding:"10px 24px",fontSize:14,fontWeight:"bold"}}>次のプレイヤーへ →</button>

                  : <button onClick={startPractice} style={{...btn("#e67e22"),padding:"10px 24px",fontSize:14,fontWeight:"bold"}}>✏️ 練習問題をやる（3問）</button>

                }

              </div>

            </div>

          )}

          {/* 練習問題 */}

          {phase==="practice"&&curPQ&&(

            <div style={{background:"#fef9e7",borderRadius:12,padding:14,animation:"fadeIn 0.4s",border:"2px solid #f39c12"}}>

                <div style={{fontSize:10,color:"#e67e22",fontWeight:"bold",marginBottom:2}}>🇯🇵 日本語</div>

                <div style={{fontSize:16,fontWeight:"bold",color:"#2c3e50",lineHeight:1.6}}>{curPQ.question}</div>

              </div>

              {curPQ.question_es&&(

                <div style={{background:"#f0f8ff",borderRadius:6,padding:"6px 10px",marginBottom:8,borderLeft:"3px solid #3498db"}}>

                  <div style={{fontSize:10,color:"#3498db",fontWeight:"bold",marginBottom:2}}>🇪🇸 Español</div>

                </div>

              )}

              <div style={{fontSize:11,color:"#7f8c8d",marginBottom:8}}>💡 {curPQ.hint}</div>

              {!practiceFeedback?(

                <div>

                  <div style={{display:"flex",gap:8}}>

                    <input ref={practiceRef} value={practiceAnswer} type="text" onChange={e=>setPracticeAnswer(e.target.value)} onKeyDown={e=>e.key==="Enter"&&submitPractice()} placeholder="こたえ…" style={{flex:1,padding:8,fontSize:16,border:"2px solid #f39c12",borderRadius:8,outline:"none"}}/>

                    <button onClick={submitPractice} style={{...btn("#e67e22"),padding:"8px 16px",fontSize:14,fontWeight:"bold"}}>こたえる</button>

                  </div>

                  <button onClick={()=>speak(curPQ.question)} style={{...btn("#9b59b6"),marginTop:8,fontSize:11,padding:"4px 10px"}}>🔊 もう一度</button>

                </div>

              ):(

                <div>

                  <div style={{fontSize:16,fontWeight:"bold",color:practiceFeedback.correct?"#27ae60":"#e74c3c",marginBottom:8}}>

                    {practiceFeedback.correct?"✅ せいかい！":`❌ ざんねん！正解は ${practiceFeedback.answer}`}

                  </div>

                  {!practiceFeedback.correct&&(

                    <div style={{background:"#fff",borderRadius:8,padding:"8px 10px",marginBottom:8,border:"1px solid #f5c6cb"}}>

                      <div style={{fontSize:11,fontWeight:"bold",color:"#c0392b",marginBottom:4}}>📖 かいせつ</div>

                      <div style={{background:"#fff8f0",borderRadius:6,padding:"5px 8px",marginBottom:5,borderLeft:"3px solid #e67e22"}}>

                        <div style={{fontSize:10,color:"#e67e22",fontWeight:"bold",marginBottom:2}}>🇯🇵 日本語</div>

                        <div style={{fontSize:12,color:"#333",lineHeight:1.6}}>{practiceFeedback.explanation}</div>

                      </div>

                      {practiceFeedback.explanation_es&&(

                        <div style={{background:"#f0f8ff",borderRadius:6,padding:"5px 8px",marginBottom:5,borderLeft:"3px solid #3498db"}}>

                          <div style={{fontSize:10,color:"#3498db",fontWeight:"bold",marginBottom:2}}>🇪🇸 Español</div>

                          <div style={{fontSize:12,color:"#333",lineHeight:1.6}}>{practiceFeedback.explanation_es}</div>

                        </div>

                      )}

                      {practiceFeedback.svg&&<ExplanationFigure svgStr={practiceFeedback.svg}/>}

                    </div>

                  )}

                  <div style={{textAlign:"center"}}>

                    <button onClick={nextPractice} style={{...btn(practiceIndex+1<practiceQuestions.length?"#e67e22":"#27ae60"),padding:"10px 24px",fontSize:14,fontWeight:"bold"}}>

                      {practiceIndex+1<practiceQuestions.length?"次の練習問題 →":"✅ 練習おわり！元のマスへ"}

                    </button>

                  </div>

                </div>

              )}

            </div>

          )}

        </div>

      );

    }
